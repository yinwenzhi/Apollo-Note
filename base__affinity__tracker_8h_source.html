68  virtual bool Init() = 0;

70  // @brief Choose some entries in the matrix to do computation for affinity
71  //
72  // Skip the entry with
73  // 1. 0 or negative affinity score
74  // 2. High score (>= 9.0)
75  //
76  // Input must be regular and of the size in the following calls
77  // (There must be the same number of entries in each row of the matrix)
78  //
79  virtual bool SelectEntries(
80  const std::vector<std::vector<float>> &prev_matrix) {
81  if (prev_matrix.empty()) return true;
82 
83  int row = prev_matrix.size();
84  if (!row) return true;
85  int col = prev_matrix[0].size();
86 
87  selected_entry_matrix_.clear();
88  selected_entry_matrix_ =
89  std::vector<std::vector<bool>>(row, std::vector<bool>(col, false));
90  for (int i = 0; i < row; ++i) {
91  for (int j = 0; j < col; ++j) {
92  if (9.0f > prev_matrix[i][j] &&
93  prev_matrix[i][j] > std::numeric_limits<float>::epsilon()) {
94  selected_entry_matrix_[i][j] = true;
95  }
96  }
97  }
98 
99  return true;
100  }
101 
102  // @brief Set the selection matrix to full entries
103  virtual bool SelectFull(const int row, const int col) {
104  selected_entry_matrix_.clear();
105  selected_entry_matrix_ =
106  std::vector<std::vector<bool>>(row, std::vector<bool>(col, true));
107  return true;
108  }
109 
110  // @brief Get affinity_matrix between tracked objs and detected objs
111  // rows: tracked objects, cols: detected objects
112  // affinity_matrix[i][j]: score for the affinity between i_th tracked obj and
113  // j_th detected obj
114  virtual bool GetAffinityMatrix(
115  const cv::Mat &img, const std::vector<Tracked> &tracked,
116  const std::vector<Detected> &detected,
117  std::vector<std::vector<float>> *affinity_matrix) = 0;
118 
119  // @brief Update information used in tracked objects, for the next frame
120  virtual bool UpdateTracked(const cv::Mat &img,
121  const std::vector<Detected> &detected,
122  std::vector<Tracked> *tracked) = 0;
123 
124  protected:
125  // @brief Matrix for selecting which entries need calculation
126  std::vector<std::vector<bool>> selected_entry_matrix_;
127 };

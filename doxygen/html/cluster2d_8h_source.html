73  scale_ = 0.5 * static_cast<float>(rows_) / range_;

75  inv_res_y_ = 0.5 * static_cast<float>(rows_) / range_;
76  point2grid_.clear();

79  pc_ptr_.reset();

84  void Cluster(const caffe::Blob<float>& category_pt_blob,
85  const caffe::Blob<float>& instance_pt_blob,
86  apollo::perception::pcl_util::PointCloudPtr pc_ptr,
87  const apollo::perception::pcl_util::PointIndices& valid_indices,
88  float objectness_thresh, bool use_all_grids_for_clustering) {
89  const float* category_pt_data = category_pt_blob.cpu_data();
90  const float* instance_pt_x_data = instance_pt_blob.cpu_data();
91  const float* instance_pt_y_data =
92  instance_pt_blob.cpu_data() + instance_pt_blob.offset(0, 1);
93 
94  pc_ptr_ = pc_ptr;
95  std::vector<std::vector<Node>> nodes(rows_,
96  std::vector<Node>(cols_, Node()));
97 
98  // map points into grids
99  size_t tot_point_num = pc_ptr_->size();
100  valid_indices_in_pc_ = &(valid_indices.indices);
101  CHECK_LE(valid_indices_in_pc_->size(), tot_point_num);
102  point2grid_.assign(valid_indices_in_pc_->size(), -1);
103 
104  for (size_t i = 0; i < valid_indices_in_pc_->size(); ++i) {
105  int point_id = valid_indices_in_pc_->at(i);
106  CHECK_GE(point_id, 0);
107  CHECK_LT(point_id, static_cast<int>(tot_point_num));
108  const auto& point = pc_ptr_->points[point_id];
109  // * the coordinates of x and y have been exchanged in feature generation
110  // step,
111  // so we swap them back here.
112  int pos_x = F2I(point.y, range_, inv_res_x_); // col
113  int pos_y = F2I(point.x, range_, inv_res_y_); // row
114  if (IsValidRowCol(pos_y, pos_x)) {
115  // get grid index and count point number for corresponding node
116  point2grid_[i] = RowCol2Grid(pos_y, pos_x);
117  nodes[pos_y][pos_x].point_num++;
118  }
119  }
120 
121  // construct graph with center offset prediction and objectness
122  for (int row = 0; row < rows_; ++row) {
123  for (int col = 0; col < cols_; ++col) {
124  int grid = RowCol2Grid(row, col);
125  Node* node = &nodes[row][col];
126  DisjointSetMakeSet(node);
127  node->is_object =
128  (use_all_grids_for_clustering || nodes[row][col].point_num > 0) &&
129  (*(category_pt_data + grid) >= objectness_thresh);
130  int center_row = std::round(row + instance_pt_x_data[grid] * scale_);
131  int center_col = std::round(col + instance_pt_y_data[grid] * scale_);
132  center_row = std::min(std::max(center_row, 0), rows_ - 1);
133  center_col = std::min(std::max(center_col, 0), cols_ - 1);
134  node->center_node = &nodes[center_row][center_col];
135  }
136  }
137 
138  // traverse nodes
139  for (int row = 0; row < rows_; ++row) {
140  for (int col = 0; col < cols_; ++col) {
141  Node* node = &nodes[row][col];
142  if (node->is_object && node->traversed == 0) {
143  Traverse(node);
144  }
145  }
146  }
147  for (int row = 0; row < rows_; ++row) {
148  for (int col = 0; col < cols_; ++col) {
149  Node* node = &nodes[row][col];
150  if (!node->is_center) {
151  continue;
152  }
153  for (int row2 = row - 1; row2 <= row + 1; ++row2) {
154  for (int col2 = col - 1; col2 <= col + 1; ++col2) {
155  if ((row2 == row || col2 == col) && IsValidRowCol(row2, col2)) {
156  Node* node2 = &nodes[row2][col2];
157  if (node2->is_center) {
158  DisjointSetUnion(node, node2);
159  }
160  }
161  }
162  }
163  }
164  }
165 
166  int count_obstacles = 0;
167  obstacles_.clear();
168  id_img_.assign(grids_, -1);
169  for (int row = 0; row < rows_; ++row) {
170  for (int col = 0; col < cols_; ++col) {
171  Node* node = &nodes[row][col];
172  if (!node->is_object) {
173  continue;
174  }
175  Node* root = DisjointSetFind(node);
176  if (root->obstacle_id < 0) {
177  root->obstacle_id = count_obstacles++;
178  CHECK_EQ(static_cast<int>(obstacles_.size()), count_obstacles - 1);
179  obstacles_.push_back(Obstacle());
180  }
181  int grid = RowCol2Grid(row, col);
182  CHECK_GE(root->obstacle_id, 0);
183  id_img_[grid] = root->obstacle_id;
184  obstacles_[root->obstacle_id].grids.push_back(grid);
185  }
186  }
187  CHECK_EQ(static_cast<size_t>(count_obstacles), obstacles_.size());
188  }

190  void Filter(const caffe::Blob<float>& confidence_pt_blob,
191  const caffe::Blob<float>& height_pt_blob) {
192  const float* confidence_pt_data = confidence_pt_blob.cpu_data();
193  const float* height_pt_data = height_pt_blob.cpu_data();
194  for (size_t obstacle_id = 0; obstacle_id < obstacles_.size();
195  obstacle_id++) {
196  Obstacle* obs = &obstacles_[obstacle_id];
197  CHECK_GT(obs->grids.size(), 0);
198  double score = 0.0;
199  double height = 0.0;
200  for (int grid : obs->grids) {
201  score += static_cast<double>(confidence_pt_data[grid]);
202  height += static_cast<double>(height_pt_data[grid]);
203  }
204  obs->score = score / static_cast<double>(obs->grids.size());
205  obs->height = height / static_cast<double>(obs->grids.size());
206  obs->cloud.reset(new apollo::perception::pcl_util::PointCloud);
207  }
208  }

210  void Classify(const caffe::Blob<float>& classify_pt_blob) {
211  const float* classify_pt_data = classify_pt_blob.cpu_data();
212  int num_classes = classify_pt_blob.channels();
213  CHECK_EQ(num_classes, static_cast<int>(MetaType::MAX_META_TYPE));
214  for (size_t obs_id = 0; obs_id < obstacles_.size(); obs_id++) {
215  Obstacle* obs = &obstacles_[obs_id];
216  for (size_t grid_id = 0; grid_id < obs->grids.size(); grid_id++) {
217  int grid = obs->grids[grid_id];
218  for (int k = 0; k < num_classes; k++) {
219  obs->meta_type_probs[k] += classify_pt_data[k * grids_ + grid];
220  }
221  }
222  int meta_type_id = 0;
223  for (int k = 0; k < num_classes; k++) {
224  obs->meta_type_probs[k] /= obs->grids.size();
225  if (obs->meta_type_probs[k] > obs->meta_type_probs[meta_type_id]) {
226  meta_type_id = k;
227  }
228  }
229  obs->meta_type = static_cast<MetaType>(meta_type_id);
230  }
231  }

233  void GetObjects(const float confidence_thresh, const float height_thresh,
234  const int min_pts_num,
235  std::vector<std::shared_ptr<Object>>* objects) {
236  CHECK(valid_indices_in_pc_ != nullptr);
237 
238  for (size_t i = 0; i < point2grid_.size(); ++i) {
239  int grid = point2grid_[i];
240  if (grid < 0) {
241  continue;
242  }
243 
244  CHECK_GE(grid, 0);
245  CHECK_LT(grid, grids_);
246  int obstacle_id = id_img_[grid];
247 
248  int point_id = valid_indices_in_pc_->at(i);
249  CHECK_GE(point_id, 0);
250  CHECK_LT(point_id, static_cast<int>(pc_ptr_->size()));
251 
252  if (obstacle_id >= 0 &&
253  obstacles_[obstacle_id].score >= confidence_thresh) {
254  if (height_thresh < 0 ||
255  pc_ptr_->points[point_id].z <=
256  obstacles_[obstacle_id].height + height_thresh) {
257  obstacles_[obstacle_id].cloud->push_back(pc_ptr_->points[point_id]);
258  }
259  }
260  }
261 
262  for (size_t obstacle_id = 0; obstacle_id < obstacles_.size();
263  obstacle_id++) {
264  Obstacle* obs = &obstacles_[obstacle_id];
265  if (static_cast<int>(obs->cloud->size()) < min_pts_num) {
266  continue;
267  }
268  std::shared_ptr<Object> out_obj(new apollo::perception::Object);
269  out_obj->cloud = obs->cloud;
270  out_obj->score = obs->score;
271  out_obj->score_type = PerceptionObstacle::CONFIDENCE_CNN;
272  out_obj->type = GetObjectType(obs->meta_type);
273  out_obj->type_probs = GetObjectTypeProbs(obs->meta_type_probs);
274  objects->push_back(out_obj);
275  }
276  }

311  void Traverse(Node* x) {
312  std::vector<Node*> p;
313  p.clear();
314  while (x->traversed == 0) {
315  p.push_back(x);
316  x->traversed = 2;
317  x = x->center_node;
318  }
319  if (x->traversed == 2) {
320  for (int i = static_cast<int>(p.size()) - 1; i >= 0 && p[i] != x; i--) {
321  p[i]->is_center = true;
322  }
323  x->is_center = true;
324  }
325  for (size_t i = 0; i < p.size(); i++) {
326  Node* y = p[i];
327  y->traversed = 1;
328  y->parent = x->parent;
329  }
330  }

332  ObjectType GetObjectType(const MetaType meta_type_id) {
333  switch (meta_type_id) {
334  case MetaType::META_UNKNOWN:
335  return ObjectType::UNKNOWN;
336  case MetaType::META_SMALLMOT:
337  return ObjectType::VEHICLE;
338  case MetaType::META_BIGMOT:
339  return ObjectType::VEHICLE;
340  case MetaType::META_NONMOT:
341  return ObjectType::BICYCLE;
342  case MetaType::META_PEDESTRIAN:
343  return ObjectType::PEDESTRIAN;
344  default: {

346  return ObjectType::UNKNOWN;

362  object_type_probs[static_cast<int>(ObjectType::PEDESTRIAN)] =
363  meta_type_probs[static_cast<int>(MetaType::META_PEDESTRIAN)];
364  return object_type_probs;

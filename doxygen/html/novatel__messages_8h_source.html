523 inline ImuParameter GetImuParameter(ImuType type) {
524  switch (type) {
525  case ImuType::IMAR_FSAS:
526  // 0.1 * (2 ** -8) * (math.pi / 180 / 3600), (0.05 * (2 ** -15)
527  return {1.893803441835e-9, 1.52587890625e-6, 200.0};
528 
529  case ImuType::ADIS16488:
530  // 720/2**31 deg/LSB, 200/2**31 m/s/LSB
531  return {5.8516723170686385e-09, 9.31322574615478515625e-8, 200.0};
532 
533  case ImuType::STIM300:
534  // 2**-21 deg/LSB, 2**-22 m/s/LSB
535  return {8.32237840649762e-09, 2.384185791015625e-07, 125.0};
536 
537  case ImuType::ISA100:
538  case ImuType::ISA100C:
539  // 1.0e-9 rad/LSB, 2.0e-8 m/s/LSB
540  return {1.0e-9, 2.0e-8, 200.0};
541 
542  case ImuType::ISA100_400HZ:
543  case ImuType::ISA100C_400HZ:
544  return {1.0e-9, 2.0e-8, 400.0};
545 
546  case ImuType::G320N:
547  return {1.7044230976507124e-11, 2.3929443359375006e-10, 125.0};
548 
549  case ImuType::CPT_XW5651:
550  return {1.0850694444444445e-07, 1.52587890625e-06, 100.0};

552  case ImuType::UM442:

35 // A navigation path tuple.
36 //
37 // first element: original navigation line index of the current navigation path.
38 // A negative value indicates illegal.
39 //
40 // second element: half of the lateral distance to the left adjacent navigation
41 // path, that is, the left width of the lane generated by this navigation path.
42 // If the navigation path is generated based on lane markers, the value is the
43 // perceived left lane width. If there is no left adjacent navigation path, the
44 // value is "default_left_width_". A negative value indicates illegal.
45 //
46 // third element: half of the lateral distance to the right adjacent navigation
47 // path, that is, the right width of the lane generated by this navigation path.
48 // If the navigation path is generated based on lane markers, the value is the
49 // perceived right lane width. If there is no right adjacent navigation path,
50 // the value is "default_right_width_". A negative value indicates illegal.
51 //
52 // fourth element : a shared pointer of the current navigation path.
53 typedef std::tuple<int, double, double, std::shared_ptr<NavigationPath>>
54  NaviPathTuple;
55 
56 // A projection index pair.
57 // pair.first: projection index of the vehicle in the current navigation line.
58 // pair.second: the distance between the vehicle's initial position and the
59 // projection position in the current navigation line.
60 typedef std::pair<int, double> ProjIndexPair;

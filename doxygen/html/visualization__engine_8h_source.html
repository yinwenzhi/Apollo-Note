221  inline void QuaternionToEuler(const double quaternion[4], double att[3]) {
222  double dcm21 =
223  2 * (quaternion[2] * quaternion[3] + quaternion[0] * quaternion[1]);
224  double dcm20 =
225  2 * (quaternion[1] * quaternion[3] - quaternion[0] * quaternion[2]);
226  double dcm22 =
227  quaternion[0] * quaternion[0] - quaternion[1] * quaternion[1] -
228  quaternion[2] * quaternion[2] + quaternion[3] * quaternion[3];
229  double dcm01 =
230  2 * (quaternion[1] * quaternion[2] - quaternion[0] * quaternion[3]);
231  double dcm11 =
232  quaternion[0] * quaternion[0] - quaternion[1] * quaternion[1] +
233  quaternion[2] * quaternion[2] - quaternion[3] * quaternion[3];
234 
235  att[0] = asin(dcm21); // the angle rotate respect to X
236  att[1] = atan2(-dcm20, dcm22); // the angle rotate respect to Y
237  att[2] = atan2(dcm01, dcm11); // the angle rotate respect to Z
238 
239  return;
240  }

145 template <typename T>
146 void uniform_slice(const T start, const T end, uint32_t num,
147  std::vector<T>* sliced) {
148  if (!sliced || num == 0) {
149  return;
150  }
151  const T delta = (end - start) / num;
152  sliced->resize(num + 1);
153  T s = start;
154  for (uint32_t i = 0; i < num; ++i, s += delta) {
155  sliced->at(i) = s;
156  }
157  sliced->at(num) = end;
158 }

171 std::unordered_set<T> Intersection(const std::unordered_set<T>& s1,
172  const std::unordered_set<T>& s2) {
173  if (s1.size() < s2.size()) {
174  std::unordered_set<T> result;

181  } else {

236 template <typename T>
237 void QuaternionToRotationMatrix(const T* quat, T* R) {
238  T x2 = quat[0] * quat[0];
239  T xy = quat[0] * quat[1];
240  T rx = quat[3] * quat[0];
241  T y2 = quat[1] * quat[1];
242  T yz = quat[1] * quat[2];
243  T ry = quat[3] * quat[1];
244  T z2 = quat[2] * quat[2];
245  T zx = quat[2] * quat[0];
246  T rz = quat[3] * quat[2];
247  T r2 = quat[3] * quat[3];
248  R[0] = r2 + x2 - y2 - z2; // fill diagonal terms
249  R[4] = r2 - x2 + y2 - z2;
250  R[8] = r2 - x2 - y2 + z2;
251  R[3] = 2 * (xy + rz); // fill off diagonal terms
252  R[6] = 2 * (zx - ry);
253  R[7] = 2 * (yz + rx);
254  R[1] = 2 * (xy - rz);
255  R[2] = 2 * (zx + ry);
256  R[5] = 2 * (yz - rx);
257 }

185 inline void L2Norm(int feat_dim, float *feat_data) {
186  if (feat_dim == 0) {
187  return;
188  }
189  // feature normalization
190  float l2norm = 0.0;
191  for (int i = 0; i < feat_dim; ++i) {
192  l2norm += feat_data[i] * feat_data[i];
193  }
194  if (l2norm == 0) {
195  float val = 1.0 / std::sqrt(feat_dim);
196  for (int i = 0; i < feat_dim; ++i) {
197  feat_data[i] = val;
198  }
199  } else {
200  l2norm = std::sqrt(l2norm);
201  for (int i = 0; i < feat_dim; ++i) {
202  feat_data[i] /= l2norm;
203  }
204  }
205 }

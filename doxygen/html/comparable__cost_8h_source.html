46  int CompareTo(const ComparableCost &other) const {
47  for (size_t i = 0; i < cost_items.size(); ++i) {
48  if (cost_items[i]) {
49  if (other.cost_items[i]) {
50  continue;
51  } else {
52  return 1;
53  }
54  } else {
55  if (other.cost_items[i]) {

57  } else {
58  continue;
59  }
60  }
61  }

65  other.smoothness_cost;
66  if (std::fabs(diff) < kEpsilon) {
67  return 0;

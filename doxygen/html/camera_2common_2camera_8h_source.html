293  virtual Eigen::Matrix<T, 2, 1> pixel_normalize(
294  const Eigen::Matrix<T, 2, 1>& pt2d) const {
295  Eigen::Matrix<T, 2, 1> pt2d_distort = CameraModel<T>::pixel_normalize(pt2d);
296 
297  Eigen::Matrix<T, 2, 1> pt2d_undistort = pt2d_distort; // Initial guess
298  for (unsigned int i = 0; i < 20; ++i) {
299  T r_sq = pt2d_undistort[0] * pt2d_undistort[0] +
300  pt2d_undistort[1] * pt2d_undistort[1];
301  T k_radial = 1.0 + distort_params_[0] * r_sq +
302  distort_params_[1] * r_sq * r_sq +
303  distort_params_[4] * r_sq * r_sq * r_sq;
304  T delta_x_0 =
305  2 * distort_params_[2] * pt2d_undistort[0] * pt2d_undistort[1] +
306  distort_params_[3] *
307  (r_sq + 2 * pt2d_undistort[0] * pt2d_undistort[0]);
308  T delta_x_1 =
309  distort_params_[2] *
310  (r_sq + 2 * pt2d_undistort[1] * pt2d_undistort[1]) +
311  2 * distort_params_[3] * pt2d_undistort[0] * pt2d_undistort[1];
312  pt2d_undistort[0] = (pt2d_distort[0] - delta_x_0) / k_radial;
313  pt2d_undistort[1] = (pt2d_distort[1] - delta_x_1) / k_radial;
314  }
315  return pt2d_undistort;
316  }

321  virtual Eigen::Matrix<T, 2, 1> pixel_denormalize(
322  const Eigen::Matrix<T, 2, 1>& pt2d) const {
323  // Add distortion
324  T r_sq = pt2d[0] * pt2d[0] + pt2d[1] * pt2d[1];
325  Eigen::Matrix<T, 2, 1> pt2d_radial =
326  pt2d *
327  (1 + distort_params_[0] * r_sq + distort_params_[1] * r_sq * r_sq +
328  distort_params_[4] * r_sq * r_sq * r_sq);
329  Eigen::Matrix<T, 2, 1> dpt2d;
330  dpt2d[0] = 2 * distort_params_[2] * pt2d[0] * pt2d[1] +
331  distort_params_[3] * (r_sq + 2 * pt2d[0] * pt2d[0]);
332  dpt2d[1] = distort_params_[2] * (r_sq + 2 * pt2d[1] * pt2d[1]) +
333  2 * distort_params_[3] * pt2d[0] * pt2d[1];
334 
335  Eigen::Matrix<T, 2, 1> pt2d_undistort;
336  pt2d_undistort[0] = pt2d_radial[0] + dpt2d[0];
337  pt2d_undistort[1] = pt2d_radial[1] + dpt2d[1];
338  // Add intrinsic K
339  return CameraModel<T>::pixel_denormalize(pt2d_undistort);
340  }

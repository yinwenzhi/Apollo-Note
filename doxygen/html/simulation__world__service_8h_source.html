220  template <typename Points>
221  void DownsampleSpeedPointsByInterval(const Points &points,
222  size_t downsampleInterval,
223  Points *downsampled_points) {
224  if (points.size() == 0) {
225  return;
226  }
227 
228  for (int i = 0; i < points.size() - 1; i += downsampleInterval) {
229  auto *point = downsampled_points->Add();
230  point->set_s(points[i].s());
231  point->set_t(points[i].t());
232  point->set_v(points[i].v());
233  }
234 
235  // add the last point
236  auto *point = downsampled_points->Add();
237  point->set_s(points[points.size() - 1].s());
238  point->set_t(points[points.size() - 1].t());
239  point->set_v(points[points.size() - 1].v());
240  }

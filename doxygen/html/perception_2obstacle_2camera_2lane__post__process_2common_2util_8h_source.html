36 // @brief: fit polynomial function with QR decomposition (using Eigen 3)
37 template <typename T = ScalarType>
38 bool PolyFit(const std::vector<Eigen::Matrix<T, 2, 1>> &pos_vec,
39  const int order, Eigen::Matrix<T, MAX_POLY_ORDER + 1, 1> *coeff,
40  const bool &is_x_axis = true) {
41  if (coeff == NULL) {
42  AERROR << "The coefficient pointer is NULL.";
43  return false;
44  }
45 
46  if (order > MAX_POLY_ORDER) {
47  AERROR << "The order of polynomial must be smaller than " << MAX_POLY_ORDER;
48  return false;
49  }
50 
51  int n = static_cast<int>(pos_vec.size());
52  if (n <= order) {
53  AERROR << "The number of points should be larger than the order. #points = "
54  << pos_vec.size();
55  return false;
56  }
57 
58  // create data matrix
59  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> A(n, order + 1);
60  Eigen::Matrix<T, Eigen::Dynamic, 1> y(n);
61  Eigen::Matrix<T, Eigen::Dynamic, 1> result;
62  for (int i = 0; i < n; ++i) {
63  float base = is_x_axis ? pos_vec[i].x() : pos_vec[i].y();
64  float p_b_j = 1.0;
65  for (int j = 0; j <= order; ++j) {
66  A(i, j) = p_b_j;
67  p_b_j *= base;
68  }
69  y(i) = is_x_axis ? pos_vec[i].y() : pos_vec[i].x();
70  }
71 
72  // solve linear least squares
73  result = A.householderQr().solve(y);
74  assert(result.size() == order + 1);
75 
76  for (int j = 0; j <= MAX_POLY_ORDER; ++j) {
77  (*coeff)(j) = (j <= order) ? result(j) : static_cast<T>(0);
78  }
79 
80  return true;
81 }

83 // @brief: evaluate y value of given x for a polynomial function
84 template <typename T = ScalarType>
85 T PolyEval(const T &x, const int order,
86  const Eigen::Matrix<T, MAX_POLY_ORDER + 1, 1> &coeff) {
87  int poly_order = order;
88  if (order > MAX_POLY_ORDER) {
89  AERROR << "the order of polynomial function must be smaller than "
90  << MAX_POLY_ORDER;
91  AINFO << "forcing polynomial order to " << MAX_POLY_ORDER;
92  poly_order = MAX_POLY_ORDER;
93  }
94 
95  T y = static_cast<T>(0);
96  float p_x_j = 1.0;
97  for (int j = 0; j <= poly_order; ++j) {
98  y += coeff(j) * p_x_j;
99  p_x_j *= x;
100  }
101 
102  return y;
103 }

105 // @brief: evaluating y value of given x for a third-order polynomial function
106 template <typename T = float>
107 T GetPolyValue(T a, T b, T c, T d, T x) {
108  T y = d;
109  T v = x;
110  y += (c * v);
111  v *= x;
112  y += (b * v);
113  v *= x;
114  y += (a * v);
115  return y;

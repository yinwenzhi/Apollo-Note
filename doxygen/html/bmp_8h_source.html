48 inline void save_rgba_image_to_bmp(const unsigned char* rgba_image, const int w,
49  const int h, const std::string& fileName) {
50  int bytes_per_line = 0;
51 
52  struct BMPHeader bmph;
53 
54  /* The length of each line must be a multiple of 4 bytes */
55  int width = w;
56  int height = h;
57  int mod4 = (width * 3) % 4;
58  if (mod4 == 0) {
59  bytes_per_line = 3 * width;
60  } else {
61  bytes_per_line = 3 * width + 4 - mod4;
62  }
63 
64  bmph.bf_type[0] = 'B';
65  bmph.bf_type[1] = 'M';
66  bmph.bf_off_bits = 54;
67  bmph.bf_size = bmph.bf_off_bits + bytes_per_line * height;
68  bmph.bf_reserved = 0;
69  bmph.bi_size = 40;
70  bmph.bi_width = width;
71  bmph.bi_height = height;
72  bmph.bi_planes = 1;
73  bmph.bi_bit_count = 24;
74  bmph.bi_compression = 0;
75  bmph.bi_size_image = bytes_per_line * height;
76  bmph.bi_x_pels_per_meter = 0;
77  bmph.bi_y_pels_per_meter = 0;
78  bmph.bi_clr_used = 0;
79  bmph.bi_clr_important = 0;
80 
81  std::fstream fs;
82  fs.open(fileName, std::fstream::in | std::fstream::binary);
83 
84  fs << bmph.bf_type;
85  fs << bmph.bf_size;
86  fs << bmph.bf_reserved;
87  fs << bmph.bf_off_bits;
88  fs << bmph.bi_size;
89  fs << bmph.bi_width;
90  fs << bmph.bi_height;
91  fs << bmph.bi_planes;
92  fs << bmph.bi_bit_count;
93  fs << bmph.bi_compression;
94  fs << bmph.bi_size_image;
95  fs << bmph.bi_x_pels_per_meter;
96  fs << bmph.bi_y_pels_per_meter;
97  fs << bmph.bi_clr_used;
98  fs << bmph.bi_clr_important;
99 
100  std::vector<unsigned char> line(bytes_per_line);
101 
102  int index = 0;

107  fs << rgba_image[index + 2];
108  index += 4;
109  }
110  }

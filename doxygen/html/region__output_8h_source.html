74 template <typename Dtype>
75 NormalizedBBox get_region_box(const std::vector<AnchorBox> anchor_boxes,
76  const Dtype *loc_data, int i, int j, int w, int h,
77  int n, int index) {
78  NormalizedBBox bbox;
79  Dtype cx = (i + sigmoid(loc_data[index + 0])) / w;
80  Dtype cy = (j + sigmoid(loc_data[index + 1])) / h;
81  Dtype hw = exp(loc_data[index + 2]) * anchor_boxes[n].w / w * 0.5;
82  Dtype hh = exp(loc_data[index + 3]) * anchor_boxes[n].h / h * 0.5;
83  bbox.xmin = cx - hw;
84  bbox.ymin = cy - hh;
85  bbox.xmax = cx + hw;
86  bbox.ymax = cy + hh;
87  return bbox;
88 }

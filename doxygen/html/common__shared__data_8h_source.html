158 template <class M>
159 void CommonSharedData<M>::RemoveStaleData() {
160  MutexLock lock(&mutex_);
161  const uint64_t now = ::time(NULL);
162  bool has_change = false;
163  for (auto iter = data_added_time_map_.begin();
164  iter != data_added_time_map_.end();) {
165  if (now - iter->second >
166  static_cast<uint64_t>(FLAGS_shared_data_stale_time)) {
167  const size_t erase_cnt = data_map_.erase(iter->first);
168  if (erase_cnt != 1u) {
169  AWARN << "data_map_ erase cnt:" << erase_cnt << " key:" << iter->first;
170  return;
171  }
172  iter = data_added_time_map_.erase(iter);
173  ++stat_.remove_cnt;
174  has_change = true;
175  } else {
176  ++iter;
177  }
178  }
179  if (has_change) {
180  AINFO << "SharedData remove_stale_data name:" << name() << " stat:["
181  << stat_.ToString() << "]";
182  }
183 }

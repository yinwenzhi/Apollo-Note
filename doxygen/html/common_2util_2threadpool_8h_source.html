82 // the constructor just launches some amount of workers
83 inline ThreadPool::ThreadPool(size_t threads) : stop(false) {
84  for (size_t i = 0; i < threads; ++i)
85  workers.emplace_back([this] {
86  for (;;) {
87  std::function<void()> task;
88 
89  {
90  std::unique_lock<std::mutex> lock(this->queue_mutex);
91  this->condition.wait(
92  lock, [this] { return this->stop || !this->tasks.empty(); });
93  if (this->stop && this->tasks.empty()) return;
94  task = std::move(this->tasks.front());
95  this->tasks.pop();
96  }
97 
98  task();
99  }
100  });
101 }

103 // add new work item to the pool
104 template <class F, class... Args>
105 auto ThreadPool::enqueue(F&& f, Args&&... args)
106  -> std::future<typename std::result_of<F(Args...)>::type> {
107  using return_type = typename std::result_of<F(Args...)>::type;
108 
109  auto task = std::make_shared<std::packaged_task<return_type()> >(
110  std::bind(std::forward<F>(f), std::forward<Args>(args)...));
111 
112  std::future<return_type> res = task->get_future();
113  {
114  std::unique_lock<std::mutex> lock(queue_mutex);
115 
116  // don't allow enqueueing after stopping the pool
117  if (stop) throw std::runtime_error("enqueue on stopped ThreadPool");
118 
119  tasks.emplace([task]() { (*task)(); });
120  }
121  condition.notify_one();
122  return res;
123 }

125 inline void ThreadPool::Synchronize() {
126  {
127  std::unique_lock<std::mutex> lock(queue_mutex);
128  stop = true;
129  }
130  condition.notify_all();
131  for (std::thread& worker : workers) worker.join();
132 }

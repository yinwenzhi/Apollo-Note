228  const PoolType* self = const_cast<const PoolType*>(this);
229  // std::recursive_mutex::scoped_lock lock(self->monitor_);
230  std::unique_lock<std::recursive_mutex> lock(self->monitor_);

329  bool execute_task() volatile {
330  std::function<void()> task;
331 
332  { // fetch task
333  PoolType* locked_this = const_cast<PoolType*>(this);
334  // std::recursive_mutex::scoped_lock lock(locked_this->monitor_);
335  std::unique_lock<std::recursive_mutex> lock(locked_this->monitor_);
336 
337  // std::cerr << "execute_task" << std::endl;
338 
339  // decrease number of threads if necessary
340  if (worker_count_ > target_worker_count_) {
341  return false; // terminate worker
342  }
343 
344  // wait for tasks
345  while (locked_this->m_scheduler.Empty()) {
346  // decrease number of workers if necessary
347  if (worker_count_ > target_worker_count_) {
348  return false; // terminate worker
349  } else {
350  active_worker_count_--;
351  locked_this->worker_idle_or_terminated_event_.notify_all();
352  locked_this->task_or_terminate_workers_event_.wait(lock);
353  active_worker_count_++;
354  // std::cerr << "active_worker_count_ ++" << std::endl;
355  }
356  }
357 
358  task = locked_this->m_scheduler.Top();
359  locked_this->m_scheduler.Pop();
360  }
361 
362  // call task function
363  if (task) {
364  task();
365  }
366 
367  // guard->Disable();
368  return true;
369  }

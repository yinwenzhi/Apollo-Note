69  template <std::size_t N>
70  std::pair<double, double> DeriveCartesianDeviation(
71  const std::size_t param_index) const {
72  auto gauss_points = common::math::GetGaussLegendrePoints<N>();
73  std::array<double, N> x = gauss_points.first;
74  std::array<double, N> w = gauss_points.second;
75 
76  std::pair<double, double> cartesian_deviation = {0.0, 0.0};
77  for (std::size_t i = 0; i < N; ++i) {
78  double r = 0.5 * x[i] + 0.5;
79  auto curr_theta = Evaluate(0, r * param_);
80  double derived_theta = DeriveTheta(param_index, r);
81 
82  cartesian_deviation.first +=
83  w[i] * (-std::sin(curr_theta)) * derived_theta;
84  cartesian_deviation.second += w[i] * std::cos(curr_theta) * derived_theta;
85  }
86 
87  cartesian_deviation.first *= param_ * 0.5;
88  cartesian_deviation.second *= param_ * 0.5;
89 
90  if (param_index == DELTA_S) {
91  for (std::size_t i = 0; i < N; ++i) {
92  double r = 0.5 * x[i] + 0.5;
93  auto theta_angle = Evaluate(0, r * param_);
94 
95  cartesian_deviation.first += 0.5 * w[i] * std::cos(theta_angle);
96  cartesian_deviation.second += 0.5 * w[i] * std::sin(theta_angle);
97  }
98  }
99  return cartesian_deviation;
100  }

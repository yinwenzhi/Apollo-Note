232 inline void ExtendedKalmanFilter<T, XN, ZN, UN>::Predict(
233  const Eigen::Matrix<T, UN, 1> &u) {

239 template<typename T, unsigned int XN, unsigned int ZN, unsigned int UN>
240 inline void ExtendedKalmanFilter<T, XN, ZN, UN>::Correct(
241  const Eigen::Matrix<T, ZN, 1> &z) {
242  y_ = z - h_(x_);
243 
244  S_ = H_ * P_ * H_.transpose() + R_;
245 
246  K_ = P_ * H_.transpose() * PseudoInverse<T, ZN>(S_);
247 
248  x_ = x_ + K_ * y_;
249 
250  P_ = (Eigen::Matrix<T, XN, XN>::Identity() - K_ * H_) * P_;
251 }

161 template <typename T>
162 bool Projector<T>::Init(const cv::Rect &roi, const T &max_distance,
163  bool visualize) {
164  AINFO << "Initialize projector ...";
165 
166  // read transformation matrix from calibration config manager
167  CalibrationConfigManager *calibration_config_manager =
168  Singleton<CalibrationConfigManager>::get();
169 
170  const CameraCalibrationPtr camera_calibration =
171  calibration_config_manager->get_camera_calibration();
172 
173  trans_mat_ = camera_calibration->get_camera2car_homography_mat().cast<T>();
174 
175  // set ROI
176  uv_xmin_ = static_cast<T>(roi.x);
177  if (uv_xmin_ < 0) {
178  AERROR << "invalid ROI x min: " << roi.x;
179  return false;
180  }
181 
182  uv_ymin_ = static_cast<T>(roi.y);
183  if (uv_ymin_ < 0) {
184  AERROR << "invalid ROI y min: " << roi.y;
185  return false;
186  }
187 
188  uv_xmax_ = static_cast<T>(roi.x + roi.width - 1);
189  if (uv_xmax_ < 0) {
190  AERROR << "invalid ROI x max: " << roi.x + roi.width - 1;
191  return false;
192  }
193 
194  uv_ymax_ = static_cast<T>(roi.y + roi.height - 1);
195  if (uv_ymax_ < 0) {
196  AERROR << "invalid ROI y max: " << roi.y + roi.height - 1;
197  return false;
198  }
199 
200  AINFO << "ROI in image space: "
201  << "x_min=" << uv_xmin_ << ", "
202  << "x_max=" << uv_xmax_ << ", "
203  << "y_min=" << uv_ymin_ << ", "
204  << "y_max=" << uv_ymax_ << ".";
205  AINFO << "ROI width = " << roi.width << ", height = " << roi.height;
206 
207  uv_roi_cols_ = static_cast<int>((uv_xmax_ - uv_xmin_) / uv_roi_x_step_) + 1;
208  uv_roi_rows_ = static_cast<int>((uv_ymax_ - uv_ymin_) / uv_roi_y_step_) + 1;
209  uv_roi_count_ = uv_roi_rows_ * uv_roi_cols_;
210  AINFO << "sampling step_x (u) = " << uv_roi_x_step_ << ", "
211  << "sampling step_y (v) = " << uv_roi_y_step_;
212  AINFO << "ROI #columns = " << uv_roi_cols_ << ", "
213  << "#rows = " << uv_roi_rows_;
214 
215  // do point-wise transform of ROI from image space to ground space
216  valid_xy_.assign(uv_roi_count_, true);
217  uv_2_xy_.resize(uv_roi_count_);
218  int count_fail_points = 0;
219  int count_too_far_points = 0;
220  T v = uv_ymin_;
221  for (int i = 0; i < uv_roi_rows_; ++i) {
222  T u = uv_xmin_;
223  for (int j = 0; j < uv_roi_cols_; ++j) {
224  // point coordinate in image space (x, y)
225  int id = i * uv_roi_cols_ + j;
226 
227  // transform point from image space to ego-car ground space
228  if (!Project(u, v, &(uv_2_xy_[id]))) {
229  valid_xy_[id] = false;
230  ++count_fail_points;
231  continue;
232  }
233 
234  // ignore points which is too far away from origin
235  if (uv_2_xy_[id].x() > max_distance || uv_2_xy_[id].y() > max_distance) {
236  valid_xy_[id] = false;
237  ++count_too_far_points;
238  continue;
239  }
240 
241  //
242  if (uv_2_xy_[id].x() < 0) {
243  valid_xy_[id] = false;
244  ++count_too_far_points;
245  continue;
246  }
247 
248  // update xy limits
249  xy_xmin_ = std::min(xy_xmin_, uv_2_xy_[id].x());
250  xy_ymin_ = std::min(xy_ymin_, uv_2_xy_[id].y());
251  xy_xmax_ = std::max(xy_xmax_, uv_2_xy_[id].x());
252  xy_ymax_ = std::max(xy_ymax_, uv_2_xy_[id].y());
253 
254  u += uv_roi_x_step_;
255  }
256  v += uv_roi_y_step_;
257  }
258  AINFO << "#failed points = " << count_fail_points << " (" << uv_roi_count_
259  << ").";
260  AINFO << "#too far points = " << count_too_far_points << " (" << uv_roi_count_
261  << ").";
262  AINFO << " xy limits: "
263  << "x_min=" << xy_xmin_ << ", "
264  << "x_max=" << xy_xmax_ << ", "
265  << "y_min=" << xy_ymin_ << ", "
266  << "y_max=" << xy_ymax_;
267 
268  is_vis_ = visualize;
269  if (is_vis_) {
270  // build a 2D grid in ground plane and
271  // find the index mapping for each valid point of image space
272  xy_image_xmin_ = std::min(std::floor(xy_xmin_), static_cast<T>(0));
273  xy_image_xmin_ = std::max(xy_image_xmin_, -max_distance);
274  xy_image_xmax_ = std::max(std::ceil(xy_xmax_), static_cast<T>(0));
275  xy_image_xmax_ = std::min(xy_image_xmax_, max_distance);
276  xy_image_ymin_ = std::min(std::floor(xy_ymin_), static_cast<T>(0));
277  xy_image_ymin_ = std::max(xy_image_ymin_, -max_distance);
278  xy_image_ymax_ = std::max(std::ceil(xy_ymax_), static_cast<T>(0));
279  xy_image_ymax_ = std::min(xy_image_ymax_, max_distance);
280 
281  x_step_ = max_distance / static_cast<T>(1000);
282  y_step_ = max_distance / static_cast<T>(1000);
283 
284  xy_image_cols_ = static_cast<int>(std::ceil(
285  (xy_image_xmax_ - xy_image_xmin_) / x_step_)) +
286  1;
287  xy_image_rows_ = static_cast<int>(std::ceil(
288  (xy_image_ymax_ - xy_image_ymin_) / y_step_)) +
289  1;
290  AINFO << "xy_image_xmin = " << xy_image_xmin_
291  << ", xy_image_xmax = " << xy_image_xmax_
292  << ", xy_image_ymin = " << xy_image_ymin_
293  << ", xy_image_ymax = " << xy_image_ymax_;
294  AINFO << "xy_image: step_x=" << x_step_ << ", "
295  << "step_y=" << y_step_;
296  AINFO << "xy_image: #cols = " << xy_image_cols_
297  << ", #rows = " << xy_image_rows_;
298 
299  xy_grid_.resize(xy_image_rows_ * xy_image_cols_);
300  T y = xy_image_ymax_;
301  for (int i = 0; i < xy_image_rows_; i++) {
302  T x = xy_image_xmin_;
303  for (int j = 0; j < xy_image_cols_; j++) {
304  xy_grid_[i * xy_image_cols_ + j].x() = x;
305  xy_grid_[i * xy_image_cols_ + j].y() = y;
306  x += x_step_;
307  }
308  y = y - y_step_;
309  }
310 
311  uv_2_xy_image_.resize(uv_roi_count_);
312  T v = uv_ymin_;
313  for (int i = 0; i < uv_roi_rows_; ++i) {
314  T u = uv_xmin_;
315  for (int j = 0; j < uv_roi_cols_; ++j) {
316  int id = i * uv_roi_cols_ + j;
317  if (valid_xy_[id]) {
318  XyToXyImagePoint(static_cast<T>(uv_2_xy_[id].x()),
319  static_cast<T>(uv_2_xy_[id].y()),
320  &uv_2_xy_image_[id]);
321  } else {
322  uv_2_xy_image_[id].x = -1;
323  uv_2_xy_image_[id].y = -1;
324  }
325  u += uv_roi_x_step_;
326  }
327  v += uv_roi_y_step_;
328  }
329  }
330 
331  is_init_ = true;
332 
333  AINFO << "succ. to initialize projector.";
334  return true;
335 }

338 bool Projector<T>::XyToXyImagePoint(const T &x, const T &y,
339  cv::Point *p) const {
340  if (!IsValidXyInXyImage(x, y)) {
341  return false;
342  }
343 
344  int j = static_cast<int>(std::round((x - xy_image_xmin_) / x_step_));
345  if (j < 0) {
346  return false;
347  }
348  if (j >= xy_image_cols_) {
349  return false;
350  }
351 
352  int i = static_cast<int>(std::round((y - xy_image_ymin_) / y_step_));
353  if (i < 0) {
354  return false;
355  }
356  if (i >= xy_image_rows_) {
357  return false;
358  }
359 
360  i = (xy_image_rows_ - 1) - i;
361 
362  p->x = j;
363  p->y = i;
364 
365  return true;
366 }

458 template <typename T>
459 bool Projector<T>::Project(const T &u, const T &v,
460  Eigen::Matrix<T, 2, 1> *xy_point) {
461  if (xy_point == nullptr) {
462  AERROR << "xy_point is a null pointer.";
463  return false;
464  }
465 
466  auto trans_mat = Singleton<CalibrationConfigManager>::get()
467  ->get_camera_calibration()
468  ->get_camera2car_homography_mat();
469 
470  Eigen::Matrix<double, 3, 1> uv_point(u, v, 1.0);
471  Eigen::Matrix<double, 3, 1> xy_p = trans_mat * uv_point;
472 
473  T scale = xy_p(2);
474  if (std::abs(scale) < 1e-6) {
475  AINFO << "Cannot solve point: scale factor is too small (" << scale << "), "
476  << " u=" << u << " v=" << v << ".";
477  return false;
478  }
479 
480  (*xy_point) << xy_p(0) / scale, xy_p(1) / scale;
481 
482  if (!std::isfinite((*xy_point)(0)) || std::isnan((*xy_point)(0)) ||
483  !std::isfinite((*xy_point)(1)) || std::isnan((*xy_point)(1))) {
484  AINFO << "xy point is not valid: "
485  << " u=" << u << " v=" << v << ".";
486  return false;
487  }
488 
489  return true;
490 }

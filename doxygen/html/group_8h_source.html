95 inline int Group::ComputeOrientation(const std::vector<Marker>& markers,
96  const GroupParam& group_param) {
97  // find valid start marker number
98  int start_marker_count = 0;
99  for (; start_marker_count < static_cast<int>(start_marker_idx.size());
100  ++start_marker_count) {
101  if (start_marker_idx[start_marker_count] == -1) {
102  break;
103  }
104  }
105  ADEBUG << "start_marker_count = " << start_marker_count;
106 
107  // compute orientation for start markers
108  switch (shape_type) {
109  case MarkerShapeType::POINT: {
110  int start_ind = 0;
111  if (start_marker_count > 1) {
112  start_ind = std::min(group_param.orientation_estimation_skip_marker_num,
113  start_marker_count - 2);
114  }
115  int end_ind = start_marker_count - 1;
116  ADEBUG << "start markers: start_ind=" << start_ind
117  << ", end_ind=" << end_ind;
118 
119  start_orie = markers.at(start_marker_idx.at(end_ind)).pos -
120  markers.at(start_marker_idx.at(start_ind)).pos;
121  break;
122  }
123 
124  case MarkerShapeType::LINE_SEGMENT: {
125  int start_ind =
126  std::min(group_param.orientation_estimation_skip_marker_num,
127  start_marker_count - 1);
128  int end_ind = start_marker_count - 1;
129  ADEBUG << "start markers: start_ind=" << start_ind
130  << ", end_ind=" << end_ind;
131 
132  start_orie = markers.at(start_marker_idx.at(end_ind)).pos -
133  markers.at(start_marker_idx.at(start_ind)).start_pos;
134  break;
135  }
136 
137  default: { AERROR << "unknown marker shape type."; }
138  }
139  start_angle = std::atan2(start_orie(1), start_orie(0));
140  RectAngle(&start_angle);
141 
142  // find valid end marker number
143  int end_marker_count = 0;
144  for (; end_marker_count < static_cast<int>(end_marker_idx.size());
145  ++end_marker_count) {
146  if (end_marker_idx[end_marker_count] == -1) {
147  break;
148  }
149  }
150  ADEBUG << "end_marker_count = " << end_marker_count;
151 
152  // compute orientation for end markers
153  switch (shape_type) {
154  case MarkerShapeType::POINT: {
155  int start_ind = end_marker_count - 1;
156  int end_ind = 0;
157  if (end_marker_count > 1) {
158  end_ind = std::min(group_param.orientation_estimation_skip_marker_num,
159  end_marker_count - 2);
160  }
161  ADEBUG << "end markers: start_ind=" << start_ind
162  << ", end_ind=" << end_ind;
163 
164  end_orie = markers.at(end_marker_idx.at(end_ind)).pos -
165  markers.at(end_marker_idx.at(start_ind)).pos;
166  break;
167  }
168 
169  case MarkerShapeType::LINE_SEGMENT: {
170  int start_ind = end_marker_count - 1;
171  int end_ind = std::min(group_param.orientation_estimation_skip_marker_num,
172  end_marker_count - 1);
173  ADEBUG << "end markers: start_ind=" << start_ind
174  << ", end_ind=" << end_ind;
175 
176  end_orie = markers.at(end_marker_idx.at(end_ind)).pos -
177  markers.at(end_marker_idx.at(start_ind)).start_pos;
178  break;
179  }
180 
181  default: { AERROR << "unknown marker shape type."; }
182  }
183  end_angle = std::atan2(end_orie(1), end_orie(0));
184  RectAngle(&end_angle);
185 
186  return end_marker_count;
187 }

189 inline ScalarType Group::ComputeDistance(const Group& tar_group,
190  const AssociationParam& param) const {
191  ADEBUG << "source point = (" << this->end_pos(0) << ", " << this->end_pos(1)
192  << ")";
193  ADEBUG << "target point = (" << tar_group.start_pos(0) << ", "
194  << tar_group.start_pos(1) << ")";
195  Vector2D displacement = tar_group.start_pos - this->end_pos;
196  ScalarType norm_v = this->end_orie.norm();
197  if (norm_v < kEpsilon) {
198  ADEBUG << "norm of orientation vector for reference group is too small: "
199  << norm_v;
200  }
201 
202  // (1) compute the projection distance of the target marker w.r.t. model
203  ScalarType projection_dist = displacement.dot(this->end_orie);
204  if (norm_v >= kEpsilon) {
205  projection_dist /= norm_v;
206  }
207  ADEBUG << "(1) projection_dist = " << std::to_string(projection_dist) << " ["
208  << std::to_string(param.min_distance) << ", "
209  << std::to_string(param.max_distance) << "]";
210  if (projection_dist < param.min_distance ||
211  projection_dist > param.max_distance) {
212  return -1;
213  }
214 
215  // (2) compute the departure distance to target marker
216  // line equation:
217  // A * x + B * y + C = 0
218  // A = v_.y(), B = -v_.x(), C = v_.x() * p_.y() - v_.y() * p_.x();
219  // the normal distance from a point (x2, y2) to the line:
220  // d = |A * x2 + B * y2 + C| / sqrt(A^2 + B^2)
221  ScalarType departure_dist =
222  std::abs(this->end_orie.y() * tar_group.start_pos.x() -
223  this->end_orie.x() * tar_group.start_pos.y() +
224  this->end_orie.x() * this->end_pos.y() -
225  this->end_orie.y() * this->end_pos.x());
226  if (norm_v >= kEpsilon) {
227  departure_dist /= norm_v;
228  }
229  ADEBUG << "(2) departure_dist = " << std::to_string(departure_dist) << " ("
230  << std::to_string(param.max_departure_distance) << ")";
231  if (departure_dist > param.max_departure_distance) {
232  return -2;
233  }
234 
235  // (3) compute the deviation angle from reference marker to the target one
236  // orientation angle of end markers on reference group
237  ScalarType beta = this->end_angle;
238  ADEBUG << "beta = " << std::to_string(beta / M_PI * 180.0);
239 
240  // angle from reference marker to the target one
241  ScalarType gamma = std::atan2(displacement(1), displacement(0));
242  if (gamma < 0) {
243  gamma += 2 * static_cast<ScalarType>(M_PI);
244  }
245  ADEBUG << "gamma = " << std::to_string(gamma / M_PI * 180.0);
246 
247  ScalarType deviation_angle_dist = std::abs(beta - gamma);
248  if (deviation_angle_dist > static_cast<ScalarType>(M_PI)) {
249  deviation_angle_dist =
250  2 * static_cast<ScalarType>(M_PI) - deviation_angle_dist;
251  }
252  ADEBUG << "(3) deviation_angle_dist = "
253  << std::to_string(deviation_angle_dist / M_PI * 180.0);
254  if (deviation_angle_dist > param.max_deviation_angle) {
255  return -3;
256  }
257 
258  // (4) relative orientation angle
259  ScalarType tar_group_start_len = tar_group.start_orie.norm();
260  ScalarType orie_dist = 0;
261  ADEBUG << "(4a) tar_group_start_len = " << std::to_string(tar_group_start_len)
262  << " (" << std::to_string(param.min_orientation_estimation_size)
263  << ")";
264  if (tar_group_start_len > param.min_orientation_estimation_size) {
265  // orientation angle of start markers on target group
266  ScalarType alpha = tar_group.start_angle;
267  ADEBUG << "alpha = " << std::to_string(alpha / M_PI * 180.0);
268 
269  orie_dist = std::abs(alpha - beta);
270  if (orie_dist > static_cast<ScalarType>(M_PI)) {
271  orie_dist = 2 * static_cast<ScalarType>(M_PI) - orie_dist;
272  }
273  ADEBUG << "(4b) orie_dist = " << std::to_string(orie_dist / M_PI * 180.0)
274  << " (" << std::to_string(param.max_relative_orie / M_PI * 180.0)
275  << ")";
276  if (orie_dist > param.max_relative_orie) {
277  return -4;
278  }
279  }
280 
281  ScalarType r =
282  std::max(std::abs(param.min_distance), std::abs(param.max_distance));
283  if (r > kEpsilon) {
284  projection_dist = std::abs(projection_dist) / r;
285  }
286  if (param.max_departure_distance > kEpsilon) {
287  departure_dist /= param.max_departure_distance;
288  }
289  if (param.max_deviation_angle > kEpsilon) {
290  deviation_angle_dist /= param.max_deviation_angle;
291  }
292  if (param.max_relative_orie > kEpsilon) {
293  orie_dist /= param.max_relative_orie;
294  }
295 
296  ScalarType dist = param.distance_weight * projection_dist +
297  param.departure_distance_weight * departure_dist +
298  param.deviation_angle_weight * deviation_angle_dist +
299  param.relative_orie_weight * orie_dist;
300 
301  ADEBUG << "overall distance = " << std::to_string(dist) << "\n";
302  return dist;
303 }

305 inline Bbox Group::GetBbox(const std::vector<Marker>& markers) const {
306  Bbox box(std::numeric_limits<ScalarType>::max(), // x_min
307  std::numeric_limits<ScalarType>::max(), // y_min
308  -std::numeric_limits<ScalarType>::max(), // x_max
309  -std::numeric_limits<ScalarType>::max()); // y_max
310 
311  for (int i : this->marker_idx) {
312  box(0) = std::min(box(0), markers[i].pos.x());
313  box(1) = std::min(box(1), markers[i].pos.y());
314  box(2) = std::max(box(2), markers[i].pos.x());
315  box(3) = std::max(box(3), markers[i].pos.y());
316 
317  if (markers[i].shape_type != MarkerShapeType::POINT) {
318  box(0) = std::min(box(0), markers[i].start_pos.x());
319  box(1) = std::min(box(1), markers[i].start_pos.y());
320  box(2) = std::max(box(2), markers[i].start_pos.x());
321  box(3) = std::max(box(3), markers[i].start_pos.y());
322  }
323  }
324 
325  return box;
326 }

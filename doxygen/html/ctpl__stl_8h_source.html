52 class Queue {
53  public:
54  bool push(const T &value) {
55  std::lock_guard<std::mutex> lock(mutex_);
56  q_.push(value);
57  return true;
58  }
59  bool push(T &&value) {
60  std::lock_guard<std::mutex> lock(mutex_);
61  q_.emplace_back(std::move(value));
62  return true;
63  }
64  // deletes the retrieved element, do not use for non integral types
65  bool pop(T &v) { // NOLINT
66  std::lock_guard<std::mutex> lock(mutex_);
67  if (q_.empty()) {
68  return false;
69  }
70  v = q_.front();
71  q_.pop();
72  return true;
73  }

76  std::lock_guard<std::mutex> lock(mutex_);
77  return q_.empty();

104  // change the number of threads in the pool
105  // should be called from one thread, otherwise be careful to not interleave,
106  // also with stop()
107  // n_threads must be >= 0
108  void Resize(const int n_threads) {
109  if (!is_stop_ && !is_done_) {
110  int old_n_threads = static_cast<int>(threads_.size());
111 
112  if (old_n_threads <=
113  n_threads) { // if the number of threads is increased
114  threads_.resize(n_threads);
115  flags_.resize(n_threads);
116  for (int i = old_n_threads; i < n_threads; ++i) {
117  flags_[i] = std::make_shared<std::atomic<bool>>(false);
118  SetThread(i);
119  }
120  } else { // the number of threads is decreased
121  for (int i = old_n_threads - 1; i >= n_threads; --i) {
122  *(flags_[i]) = true; // this thread will finish
123  threads_[i]->detach();
124  }
125 
126  // stop the detached threads that were waiting
127  cv_.notify_all();
128 
129  // safe to delete because the threads are detached
130  threads_.resize(n_threads);
131 
132  // safe to delete because the threads
133  // have copies of shared_ptr of the
134  // flags, not originals
135  flags_.resize(n_threads);
136  }
137  }

156  // wait for all computing threads to finish and stop all threads
157  // may be called asynchronously to not pause the calling thread while waiting
158  // if is_wait == true, all the functions in the queue are run, otherwise the
159  // queue is cleared without running the functions
160  void Stop(bool is_wait = false) {
161  if (!is_wait) {
162  if (is_stop_) {
163  return;
164  }
165  is_stop_ = true;
166  for (int i = 0, n = size(); i < n; ++i) {
167  *(flags_[i]) = true; // command the threads to stop
168  }
169  ClearQueue(); // empty the queue
170  } else {
171  if (is_done_ || is_stop_) return;
172  is_done_ = true; // give the waiting threads a command to finish
173  }
174 
175  cv_.notify_all(); // stop all waiting threads
176 
177  for (int i = 0; i < static_cast<int>(threads_.size());
178  ++i) { // wait for the computing threads to finish
179  if (threads_[i]->joinable()) {
180  threads_[i]->join();
181  }
182  }
183  // if there were no threads in the pool but some functions in the queue, the
184  // functions are not deleted by the threads
185  // therefore delete them here
186  ClearQueue();
187  threads_.clear();
188  flags_.clear();
189  }

191  template <typename F, typename... Rest>
192  auto Push(F &&f, Rest &&... rest) -> std::future<decltype(f(0, rest...))> {
193  auto pck =
194  std::make_shared<std::packaged_task<decltype(f(0, rest...))(int)>>(
195  std::bind(std::forward<F>(f), std::placeholders::_1,
196  std::forward<Rest>(rest)...));
197  auto _f = std::make_shared<std::function<void(int id)>>(
198  [pck](int id) { (*pck)(id); });
199  // It is not necessary to lock q_ because it is locked in the Queue class.
200  q_.push(_f);
201  cv_.notify_one();
202 
203  return pck->get_future();
204  }

230  void SetThread(int i) {
231  std::shared_ptr<std::atomic<bool>> flag(
232  flags_[i]); // a copy of the shared ptr to the flag
233  auto f = [this, i, flag /* a copy of the shared ptr to the flag */]() {
234  std::atomic<bool> &_flag = *flag;
235  std::shared_ptr<std::function<void(int id)>> _f;
236  bool is_pop_ = q_.pop(_f);
237  while (true) {
238  while (is_pop_) { // if there is anything in the queue
239  (*_f)(i);
240  if (_flag) {
241  // the thread is wanted to stop, return even if the queue is not
242  // empty yet
243  return;
244  } else {
245  is_pop_ = q_.pop(_f);
246  }
247  }
248  // the queue is empty here, wait for the next command
249  {
250  std::unique_lock<std::mutex> lock(mutex_);
251  ++n_waiting_;
252  cv_.wait(lock, [this, &_f, &is_pop_, &_flag]() {
253  is_pop_ = q_.pop(_f);
254  return is_pop_ || is_done_ || _flag;
255  });
256  --n_waiting_;
257  if (!is_pop_) {
258  // if the queue is empty and is_done_ == true or *flag
259  // then return
260  return;
261  }
262  }
263  }
264  };
265  threads_[i].reset(
266  new std::thread(f)); // compiler may not support std::make_unique()
267  }

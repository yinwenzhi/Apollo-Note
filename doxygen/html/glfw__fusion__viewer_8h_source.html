53 template <typename T = float>
54 T get_poly_value(T a, T b, T c, T d, T x) {
55  T y = d;
56  T v = x;
57  y += (c * v);
58  v *= x;
59  y += (b * v);
60  v *= x;
61  y += (a * v);
62  return y;
63 }

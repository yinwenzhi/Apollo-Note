150  void SetCar2CameraExtrinsicsAdj(Eigen::Matrix<double, 4, 4> matrix,
151  bool adjusted) {
152  std::lock_guard<std::mutex> lock(adj_mtx_);
153  camera2car_adj_ = matrix;
154  adjusted_extrinsic_ = adjusted;
155 
156  auto c_int_inv = camera_intrinsic_.block(0, 0, 3, 3).inverse();
157  auto car2camera_3_4 = (camera2car_adj_.inverse()).block(0, 0, 3, 4);
158  Eigen::Matrix3d camera_2car_stripped;
159  camera_2car_stripped.col(0) = car2camera_3_4.col(0);
160  camera_2car_stripped.col(1) = car2camera_3_4.col(1);
161  camera_2car_stripped.col(2) = car2camera_3_4.col(3);
162 
163  homography_camera2car_adj_ =
164  camera_2car_stripped.inverse() * c_int_inv;
165  }
166 
167  bool GetCar2CameraExtrinsicsAdj(Eigen::Matrix<double, 4, 4>* matrix) {
168  std::lock_guard<std::mutex> lock(adj_mtx_);
169  *matrix = camera2car_adj_;
170  return adjusted_extrinsic_;
171  }

255 template <typename T, unsigned int XN, unsigned int ZN, unsigned int UN>
256 inline void KalmanFilter<T, XN, ZN, UN>::Predict(
257  const Eigen::Matrix<T, UN, 1> &u) {
258  CHECK(is_initialized_);
259 
260  x_ = F_ * x_ + B_ * u;
261 
262  P_ = F_ * P_ * F_.transpose() + Q_;
263 }
264 
265 template <typename T, unsigned int XN, unsigned int ZN, unsigned int UN>
266 inline void KalmanFilter<T, XN, ZN, UN>::Correct(
267  const Eigen::Matrix<T, ZN, 1> &z) {
268  CHECK(is_initialized_);
269  y_ = z - H_ * x_;
270 
271  S_ = H_ * P_ * H_.transpose() + R_;
272 
273  K_ = P_ * H_.transpose() * PseudoInverse<T, ZN>(S_);
274 
275  x_ = x_ + K_ * y_;
276 
277  P_ = (Eigen::Matrix<T, XN, XN>::Identity() - K_ * H_) * P_;
278 }

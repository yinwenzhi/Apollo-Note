413  // HasSequenceNumber returns true if the data type is proto and has
414  // header.sequence_num.
415  template <typename InputMessageType>
416  static bool HasSequenceNumber(
417  enable_if_t<
418  std::is_base_of<google::protobuf::Message, InputMessageType>::value,
419  InputMessageType>* message = nullptr) {
420  using gpf = google::protobuf::FieldDescriptor;
421  InputMessageType sample;
422  auto descriptor = sample.GetDescriptor();
423  auto header_descriptor = descriptor->FindFieldByName("header");
424  if (header_descriptor == nullptr ||
425  header_descriptor->cpp_type() != gpf::CPPTYPE_MESSAGE) {
426  return false;
427  }
428 
429  auto sequence_num_descriptor =
430  header_descriptor->message_type()->FindFieldByName("sequence_num");
431  return sequence_num_descriptor != nullptr &&
432  sequence_num_descriptor->cpp_type() == gpf::CPPTYPE_UINT32;
433  }

443  // DumpMessage dumps the message to a file to
444  // /tmp/<adapter_name>/<name>.pb.txt, where the message is in ASCII
445  // mode and <name> is the .header().sequence_num() of the message.
446  template <typename InputMessageType>
447  bool DumpMessage(const enable_if_t<std::is_base_of<google::protobuf::Message,
448  InputMessageType>::value,
449  InputMessageType>& message) {
450  using google::protobuf::Message;
451  auto descriptor = message.GetDescriptor();
452  auto header_descriptor = descriptor->FindFieldByName("header");
453  if (header_descriptor == nullptr) {
454  ADEBUG << "Fail to find header field in pb.";
455  return false;
456  }
457  const Message& header = message.GetReflection()->GetMessage(
458  *static_cast<const Message*>(&message), header_descriptor);
459  auto seq_num_descriptor =
460  header.GetDescriptor()->FindFieldByName("sequence_num");
461  if (seq_num_descriptor == nullptr) {
462  ADEBUG << "Fail to find sequence_num field in pb.";
463  return false;
464  }
465  uint32_t sequence_num =
466  header.GetReflection()->GetUInt32(header, seq_num_descriptor);
467  return util::SetProtoToASCIIFile(
468  message, util::StrCat(dump_path_, "/", sequence_num, ".pb.txt"));
469  }

497  void EnqueueData(DataPtr data) {
498  if (enable_dump_) {

502  // Don't try to copy data and enqueue if the message_num is 0
503  if (message_num_ == 0) {

506

508  std::lock_guard<std::mutex> lock(mutex_);
509  if (data_queue_.size() + 1 > message_num_) {
510  data_queue_.pop_back();
511  }

513  }
